          import json
          import boto3
          import logging
          from datetime import datetime
          
          logger = logging.getLogger()
          logger.setLevel(logging.INFO)
          s3_client = boto3.client('s3')
          
          def lambda_handler(event, context):
              try:
                  logger.info(f"Checking for duplicates: {json.dumps(event, default=str)}")
                  
                  job_id = event['job_id']
                  invoice_data = event['invoice_data']
                  results_bucket = event['results_bucket']
                  
                  # Get current invoice details
                  current_invoice_id = invoice_data.get('invoice_id')
                  current_amount = invoice_data.get('amount')
                  current_vendor = invoice_data.get('vendor_name')
                  
                  duplicates_found = []
                  
                  if current_invoice_id and current_amount:
                      # List existing processed invoices
                      try:
                          response = s3_client.list_objects_v2(
                              Bucket=results_bucket,
                              Prefix='processed/',
                              MaxKeys=1000
                          )
                          
                          for obj in response.get('Contents', []):
                              if obj['Key'].endswith('final_result.json') and job_id not in obj['Key']:
                                  try:
                                      # Get existing invoice data
                                      existing_response = s3_client.get_object(
                                          Bucket=results_bucket,
                                          Key=obj['Key']
                                      )
                                      existing_data = json.loads(existing_response['Body'].read())
                                      
                                      existing_bedrock = existing_data.get('bedrock_data', {})
                                      existing_invoice_id = existing_bedrock.get('invoice_id')
                                      existing_amount = existing_bedrock.get('amount')
                                      existing_vendor = existing_bedrock.get('vendor_name')
                                      
                                      # Check for duplicates
                                      is_duplicate = False
                                      match_reasons = []
                                      
                                      if existing_invoice_id and current_invoice_id:
                                          if existing_invoice_id.lower() == current_invoice_id.lower():
                                              is_duplicate = True
                                              match_reasons.append('invoice_id_exact_match')
                                      
                                      if existing_amount and current_amount:
                                          if abs(float(existing_amount) - float(current_amount)) < 0.01:
                                              if existing_vendor and current_vendor:
                                                  if existing_vendor.lower() == current_vendor.lower():
                                                      is_duplicate = True
                                                      match_reasons.append('amount_vendor_match')
                                      
                                      if is_duplicate:
                                          duplicates_found.append({
                                              'existing_job_id': existing_data.get('job_id'),
                                              'existing_invoice_id': existing_invoice_id,
                                              'existing_amount': existing_amount,
                                              'existing_vendor': existing_vendor,
                                              'match_reasons': match_reasons,
                                              'confidence': 0.95 if 'invoice_id_exact_match' in match_reasons else 0.85
                                          })
                                          
                                  except Exception as e:
                                      logger.warning(f"Error processing {obj['Key']}: {str(e)}")
                                      continue
                                      
                      except Exception as e:
                          logger.error(f"Error listing objects: {str(e)}")
                  
                  return {
                      'success': True,
                      'job_id': job_id,
                      'duplicates_found': duplicates_found,
                      'duplicate_count': len(duplicates_found),
                      'check_timestamp': datetime.utcnow().isoformat()
                  }
                  
              except Exception as e:
                  logger.error(f"Error checking duplicates: {str(e)}")
                  return {'success': False, 'error': str(e)}

          import json
          import boto3
          import logging
          from datetime import datetime
          
          logger = logging.getLogger()
          logger.setLevel(logging.INFO)
          s3_client = boto3.client('s3')
          
          def lambda_handler(event, context):
              try:
                  logger.info(f"Generating report: {json.dumps(event, default=str)}")
                  
                  job_id = event['job_id']
                  processing_results = event['processing_results']
                  results_bucket = event['results_bucket']
                  
                  # Generate summary report
                  report = {
                      'job_id': job_id,
                      'report_timestamp': datetime.utcnow().isoformat(),
                      'processing_summary': {
                          'textract_success': processing_results.get('textract', {}).get('success', False),
                          'sagemaker_success': processing_results.get('sagemaker', {}).get('success', False),
                          'bedrock_success': processing_results.get('bedrock', {}).get('success', False),
                          'duplicate_check_success': processing_results.get('duplicate_check', {}).get('success', False)
                      },
                      'invoice_summary': {},
                      'alerts': [],
                      'recommendations': []
                  }
                  
                  # Extract key information
                  if processing_results.get('bedrock', {}).get('success'):
                      bedrock_data = processing_results['bedrock'].get('curated_invoice_data', {})
                      report['invoice_summary'] = {
                          'invoice_id': bedrock_data.get('invoice_id'),
                          'vendor': bedrock_data.get('vendor_name'),
                          'amount': bedrock_data.get('amount'),
                          'due_date': bedrock_data.get('due_date'),
                          'risk_level': bedrock_data.get('overall_risk_level'),
                          'data_quality_score': bedrock_data.get('data_quality_score')
                      }
                      
                      report['alerts'] = bedrock_data.get('alerts', [])
                      report['recommendations'] = bedrock_data.get('recommendations', [])
                  
                  # Add duplicate information
                  if processing_results.get('duplicate_check', {}).get('success'):
                      duplicate_data = processing_results['duplicate_check']
                      if duplicate_data.get('duplicate_count', 0) > 0:
                          report['alerts'].append({
                              'type': 'duplicate',
                              'severity': 'high',
                              'title': 'Duplicate Invoice Detected',
                              'description': f"Found {duplicate_data['duplicate_count']} potential duplicate(s)",
                              'details': duplicate_data.get('duplicates_found', [])
                          })
                  
                  # Store report
                  report_key = f"reports/{job_id}/processing_report.json"
                  s3_client.put_object(
                      Bucket=results_bucket,
                      Key=report_key,
                      Body=json.dumps(report, default=str, indent=2),
                      ContentType='application/json'
                  )
                  
                  return {
                      'success': True,
                      'job_id': job_id,
                      'report_location': f"s3://{results_bucket}/{report_key}",

          def lambda_handler(event, context):
              try:
                  logger.info(f"Sending notification: {json.dumps(event, default=str)}")
                  
                  job_id = event['job_id']
                  status = event['status']
                  sns_topic_arn = os.environ.get('SNS_TOPIC_ARN')
                  
                  if status == 'completed':
                      results_summary = event.get('results_summary', {})
                      alerts = event.get('alerts', [])
                      
                      # Create notification message
                      message = f"""
PayScan AI Agent - Processing Complete

Job ID: {job_id}
Status: {status}
Timestamp: {datetime.utcnow().isoformat()}

Invoice Summary:
- Invoice ID: {results_summary.get('invoice_id', 'N/A')}
- Vendor: {results_summary.get('vendor', 'N/A')}
- Amount: ${results_summary.get('amount', 'N/A')}
- Due Date: {results_summary.get('due_date', 'N/A')}
- Risk Level: {results_summary.get('risk_level', 'N/A')}

Alerts Generated: {len(alerts)}
"""
                      
                      if alerts:
                          message += "\nAlerts:\n"
                          for alert in alerts[:3]:  # Show first 3 alerts
                              message += f"- {alert.get('severity', 'Unknown').upper()}: {alert.get('title', 'Alert')}\n"
                  
                  elif status == 'failed':
                      error_type = event.get('error_type', 'unknown')
                      error_details = event.get('error_details', {})
                      
                      message = f"""
PayScan AI Agent - Processing Failed

Job ID: {job_id}
Status: {status}
Error Type: {error_type}
Timestamp: {datetime.utcnow().isoformat()}

Error Details: {json.dumps(error_details, indent=2)}
"""
                  
                  else:
                      message = f"""
PayScan AI Agent - Status Update

Job ID: {job_id}
Status: {status}
Timestamp: {datetime.utcnow().isoformat()}
"""
                  
                  # Send SNS notification
                  if sns_topic_arn:
                      sns_client.publish(
                          TopicArn=sns_topic_arn,
                          Subject=f"PayScan AI Agent - {status.title()} ({job_id})",
                          Message=message
                      )
                  
                  return {
                      'success': True,
                      'job_id': job_id,
                      'notification_sent': True
                  }
                  
              except Exception as e:
                  logger.error(f"Error sending notification: {str(e)}")
                  return {'success': False, 'error': str(e)}
